+
+        final ProgressDialog dlg = new ProgressDialog(this);
+        dlg.setTitle("Loading...");
+        dlg.setIndeterminate(false);
+        dlg.setProgressStyle(ProgressDialog.STYLE_HORIZONTAL);

+        .setCallback(new FutureCallback<ImageView>() {

+                dlg.cancel();
+            }
+        });
