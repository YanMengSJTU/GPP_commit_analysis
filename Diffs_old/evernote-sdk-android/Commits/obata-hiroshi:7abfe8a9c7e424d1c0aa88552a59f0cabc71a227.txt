+		mListView = (ListView) findViewById(R.id.notebook_listview);

+	    try {
+	        mEvernoteSession.getClientFactory().createNoteStoreClient().listNotebooks(new OnClientCallback<List<Notebook>>() {

+	        });

+	      }	
+	}

+		mAdapter.clear();
